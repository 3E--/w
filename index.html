Sample "Hello World"

#include <w>

int main(int argc, char** argv) {
  w::App app { argc, argv };

  app.get("/", [](w::Request& req) {
    return w::render_text("Hello, World!");
  });

  return app.run();
}

Save as "hello.cpp" and compile with:

clang++ -Iw -o hello hello.cpp

Run as:

./hello
